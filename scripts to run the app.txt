#For docker-compose on Google Compute Engine
docker run docker/compose:1.24.0 version

echo alias docker-compose="'"'docker run --rm \
    -v /var/run/docker.sock:/var/run/docker.sock \
    -v "$PWD:$PWD" \
    -w="$PWD" \
    docker/compose:1.24.0'"'" >> ~/.bashrc

docker stop $(docker ps -aq)
docker rm $(docker ps -aq)

docker login ... #log-in to your docker hub account

docker network create mynetwork

docker pull bmcnpnr/service-discovery
docker pull bmcnpnr/api-gateway
docker pull bmcnpnr/status-checker
docker pull bmcnpnr/status-simulator 
docker pull bmcnpnr/status-checker-webapp
docker network inspect mynetwork

docker run --name service-discovery -d -p 8761:8761 --expose=8761 --network mynetwork bmcnpnr/service-discovery
docker run --name api-gateway -d -p 8762:8762 --expose=8762 --network mynetwork bmcnpnr/api-gateway
docker run --name status-checker -d -p 8080:8080 --network mynetwork bmcnpnr/status-checker
docker run --name status-simulator -d -p 8081:8081 --network mynetwork bmcnpnr/status-simulator
docker run -d -p 5000:5000 --expose=5000 --network mynetwork bmcnpnr/status-checker-webapp
